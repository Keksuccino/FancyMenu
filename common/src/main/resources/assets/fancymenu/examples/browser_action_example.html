Example Code:

// Action without value fancymenu.execute('quitgame'); // Action with value fancymenu.execute('opengui', 'title_screen'); // Set variable with two values (name:value) fancymenu.execute('set_variable', 'example_variable:example_value'); // With callbacks (action without value) fancymenu.executeWithCallback('quitgame', function(result) { console.log('Success:', result); }, function(error) { console.error('Error:', error); } ); // With callbacks (action with value) fancymenu.executeWithCallback('opengui', 'title_screen', function(result) { console.log('GUI opened'); }, function(error) { console.error('Failed:', error); } ); // Using the global alias FancyMenu.execute('quitgame'); FancyMenu.execute('opengui', 'title_screen');
